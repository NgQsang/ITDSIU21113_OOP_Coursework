//using inherritance and  polymorphism

//parent class
public class Course {
    private String name;
    private int minEngScore;

    //constructor
    public Course(String name, int minEngScore) {
        this.name = name;
        this.minEngScore = minEngScore;
    }

    public String getCourseName() {
        return name;
    }

    public int getMinEngScore() {
        return minEngScore;
    }
}

//Student class to get the test result
public class Student {
    private int TestResult;

    public Student(int TestResult) {
        this.TestResult = TestResult;
    }

    public int getTestResult() {
        return TestResult;
    }
}

//EngCourse class
public class EngCourse extends Course {
    public EngCourse(String name, int minEngScore) {
        super(name, minEngScore);
    }
}

//PTCourse  class
public class PTCourse extends Course {
    public PTCourse(String name, int minEngScore) {
        super(name, minEngScore);
    }
}

//CourseRegistration class

class CourseRegistration {
    public boolean register(Student student, Course course) {
        int TestResult = student.getTestResult();
        String courseName = course.getCourseName();

        if (TestResult >= 80) {
            return true; // Group A can take any course
        } else if (TestResult >= 60 && TestResult <= 79) {
            if (courseName.equals("IE1") || courseName.equals("IE2") || courseName.equals("PT1") || courseName.equals("PT2")) {
                return true; // Group B can take intermediate English or physical training courses
            }
        } else if (courseName.equals("EE1") || courseName.equals("EE2")) {
            return true; // Group C can take basic English courses
        }

        return false;
    }
}

//Main class
public class Main {
    public static void main(String[] args) {
        Student studentgroupA = new Student(TestResult: 90);
        Student studentgroupB = new Student(TestResult: 75);
        Student studentgroupC = new Student(TestResult: 50);

        CourseRegistration registration = new CourseRegistration();

        Course ie1 = new EngCourse("IE1", 60);
        Course pt1 = new PTCourse("PT1", 60);

        System.out.println("Student in Group A can register for IE1: " + registration.register(studentgroupA, ie1));
        System.out.println("Student in Group B can register for PT1: " + registration.register(studentgroupB, pt1));
        System.out.println("Student in Group C can register for IE1: " + registration.register(studentgroupC, ie2));
    }
}
